package com.zensar.ide.dto;

public class CoupenDto {
	private int coupenId;
	private String expDate;
	private String coupenCode;
	private int coupenPrice;
	
	public CoupenDto() {
		super();
	}
	public CoupenDto(int coupenId, String expDate, string coupenCode, int coupenPrice) {
		super();
		this.coupenId = coupenId;
		this.expDate = expDate;
		this.coupenCode = coupenCode;
		 this.coupenPrice = coupenPrice;
	}
	public int getCoupenId() {
		return coupenId;
	}
	public void setCoupenId(int coupenId) {
		this.coupenId = coupenId;
	}

	public String getExpDate() {
		return expDate;
	}
	public void setExpDate(String expDate) {
		this.expDate = expDate;
	}
	public String getCoupenCode() {
		return coupenCode;
	}
	public void setCoupenCode(String coupenCode) {
		this.coupenCode = coupenCode;
	}
	
	public int getCoupenPrice() {
		return coupenPrice;
	}
	public void setCoupenPrice(int coupenPrice) {
		this.coupenPrice = coupenPrice;
	}

	@Override
	public String toString() {
		return "CoupenDto [coupenId=" + coupenId + ",  expDate=" + expDate + ", coupenCode=" + coupenCode + ", coupenPrice=" + coupenPrice + "]";
	}
	
}
