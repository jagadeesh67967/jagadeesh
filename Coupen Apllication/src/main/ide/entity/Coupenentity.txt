package com.zensar.ide.entity;

import javax.persistence.Entity;
import javax.persistence.Id;

@Entity
public class Coupen {
	@Id
	private int coupenId;
	private String ExpiryDate;
	private String coupenCode;
	

	public Coupen() {
		super();
	}

	public Coupen(int coupenId,String coupenName, String coupenExpiryDate, String coupenCode) {
		super();
		this.coupenId = coupenId;
		this.ExpiryDate = ExpiryDate;
		this.coupenCode = coupenCode;
	}

	public int getCoupenId() {
		return coupenId;
	}

	public void setCoupenId(int coupenId) {
		this.coupenId = coupenId;
	}

	public String getExpiryDate() {
		return ExpiryDate;
	}

	public void setExpiryDate(String ExpiryDate) {
		this.ExpiryDate = coupenExpiryDate;
	}

	public int getCoupenCode() {
		return coupenCode;
	}

	public void setCoupenPrice(String coupenCode) {
		this.coupenCode = coupenCode;
	}

	

	@Override
	public String toString() {
		return "Coupen [coupenId=" + coupenId + ", ExpiryDate=" + ExpiryDate + ", coupenCode=" + coupenCode + "]";
	}

}
