package com.zensar.ide.entity;

import javax.persistence.Entity;
import javax.persistence.Id;

@Entity
public class Coupen {
	@Id
	private int coupenId;
	private String coupenName;
	private int coupenExpiryDate;
	private String coupenCode;
	

	public Coupen() {
		super();
	}

	public Coupen(int coupenId,String coupenName, int coupenExpiryDate, String coupenCode) {
		super();
		this.coupenId = coupenId;
		this.coupenName = coupenName;
		this.coupenExpiryDate = coupenExpiryDate;
		this.coupenCode = coupenCode;
	}

	public int getCoupenId() {
		return coupenId;
	}

	public void setCoupenId(int coupenId) {
		this.coupenId = coupenId;
	}
	
	public int getCoupenName() {
		return coupenName;
	}

	public void setCoupenName(int coupenName) {
		this.coupenName = coupenName;
	}

	public String getcoupenExpiryDate() {
		return coupenExpiryDate;
	}

	public void setcoupenExpiryDate(int coupenExpiryDate) {
		this.coupenExpiryDate = coupenExpiryDate;
	}

	public int getCoupenCode() {
		return coupenCode;
	}

	public void setCoupenPrice(String coupenCode) {
		this.coupenCode = coupenCode;
	}

	

	@Override
	public String toString() {
		return "Coupen [coupenId=" + coupenId + ",coupenName=" + coupenName + ", coupenExpiryDate=" + coupenExpiryDate + ", coupenCode=" + coupenCode
				+ "]";
	}

}
