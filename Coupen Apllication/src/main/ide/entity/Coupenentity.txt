package com.zensar.ide.entity;

import javax.persistence.Entity;
import javax.persistence.Id;

@Entity
public class Coupen {
	@Id
	private int coupenId;
	private String coupenName;
	private String coupenCode;
	

	public Coupen() {
		super();
	}

	public Coupen(int coupenId, String coupenName, String coupenCode) {
		super();
		this.coupenId = coupenId;
		this.coupenName = coupenName;
		this.coupenCode = coupenCode;
	}

	public int getCoupenId() {
		return coupenId;
	}

	public void setCoupenId(int coupenId) {
		this.coupenId = coupenId;
	}

	public String getCoupenName() {
		return coupenName;
	}

	public void setCoupenName(String coupenName) {
		this.coupenName = coupenName;
	}

	public int getCoupenCode() {
		return coupenCode;
	}

	public void setCoupenPrice(String coupenCode) {
		this.coupenCode = coupenCode;
	}

	

	@Override
	public String toString() {
		return "Coupen [coupenId=" + coupenId + ", coupenName=" + coupenName + ", coupenCode=" + coupenCode
				+ "]";
	}

}
